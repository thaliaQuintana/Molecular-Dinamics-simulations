def get_match(match, line, init, end):
    #gridposition counts
    value = re.findall(match, line)[init:end]
    print(value)
    return value
#Extract information from first eight lines
def get_info(file_path):
    with open(file_path, "r") as file:
        lines = file.readlines()
    first_three_lines = lines[:8]

    counts=[int(num) for num in get_match(r'\d+',lines[1],1,4)]
    origin=[float(num) for num in get_match(r"-?\d+\.\d+",lines[2],0,4)]
    dx=[float(num) for num in get_match(r'\d+\.\d+',lines[3],0,1)][0]
    dy=[float(num) for num in get_match(r'\d+\.\d+',lines[4],0,1)][0]
    dz=[float(num) for num in get_match(r'\d+\.\d+',lines[5],0,1)][0]
    delta=[dx,dy,dy]
    return counts,origin,delta

#Get coordinates and potential
def get_coordinates_average_potential(outputname, data, counts,origin,delta,rx,cx,ry,cy):
    Nx,Ny,Nz=counts
    ox,oy,oz=origin
    dx,dy,dz=delta
    #write coordinates in .txt file
    with open(outputname,'w') as file: 
        for ind, row in enumerate(data):
            i=int(ind/(Ny*Nz))%Nx
            j=int(ind/(Nz))%Ny
            k=ind%Ny
            avg=np.mean(row)
            x=ox+dx*i
            y=oy+dy*j
            z=oz+dz*k
           
            file.write(f'{x:.2f}, {y:.2f}, {z:.2f}, {avg: .2f}\n')
            #print(i,j,k)
    pmeplot=np.loadtxt(outputname, dtype='float',delimiter=',')
    print(len(pmeplot))
    #Separate coordinates and potential in different array
    X=[]
    Y=[]
    Z=[]
    V=[]
    Zl=[]
    Vz=[]
    for ind, row in enumerate(pmeplot):

        X.append(row[0])
        Y.append(row[1])
        Z.append(row[2])
        V.append(row[3])

        if row[0]==pmeplot[rx][cx] and row[1]==pmeplot[ry][cy]: 
            Zl.append(row[2])
            Vz.append(row[3])
    return X, Y, Z, V, Zl, Vz

def plot_resultingVE(inputfilename, outputfilename, EPoutputfilename, a, unit, yprofile, counts,origin,delta,legendname): 
    
    #delete_rowsdx(inputfilename, outputfilename)
    f = open(inputfilename, "r")
    data = f.readlines()
    data = data[8:-5]
    
    #Get coordinates and potential.
    Xx, Yx, Zx, Vx, X, Y, Z =get_coordinates_potential(EPoutputfilename, data, counts,origin,delta,a,unit)
    
    Zl=[]
    Vz=[]
    #yidx=int((yprofile-origin[1])/delta[1])
    for ind, z in enumerate(Zx):
        if Yx[ind]==Y[yprofile]: 
                    Zl.append(z)
                    Vz.append(Vx[ind])
    
    #Electric potential             
    EP=plt.subplot(211)         
    #plot z profile 
    EP.plot(Zl,Vz,label =f'{legendname}')
    #plt.plot(Vz,Zl)
    #EP.set_xlabel('Z ($\AA$)')
    EP.set_ylabel('$\Psi_{z} (V)$')
    
    #Electric field
    
    EF=plt.subplot(212)
    dy_dx = np.gradient (Vz,Zl)
    EF.plot(Zl,dy_dx,label =f'{legendname}')

    EF.set_xlabel('Z ($\AA$)')
    EF.set_ylabel('E(V/$\AA)$')
